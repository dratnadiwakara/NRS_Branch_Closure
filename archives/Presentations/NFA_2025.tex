\begin{frame}{Why \textit{were} branches so important (1980-2008)?}
    \begin{itemize}
        \item Branch location affected competition (regulatory restrictions limited local-market contestability)
        \item Branch presence affected cost of information collection
        \item Extent of branch networks affected capital mobility
        \item Local branch concentration affected monetary policy transmission
    \end{itemize}
\vspace{0.5cm}
When regulatory constraints on branching were loosened 
    \begin{itemize}
        \item more competiton; bank M\& A exploded 
        \item more extenisve branch networks
        \item cheaper credit; better credit allocation
    \end{itemize}
\end{frame}

\begin{frame}{Customer adoption of technology driving transition away from branches post GFC }
\begin{itemize}
    \item Technology is ascendant
    \begin{itemize}
        \item Less need for branches
        \item Easier to move money
    \end{itemize}
    \item[]
    \item Financially sophisticated $\longrightarrow$ $\uparrow$  interest sensitive and tech savvy
    \begin{itemize}
        \item Impact of technology is stronger for these depositors
    \end{itemize}
    \item[]
    \item $\uparrow$  interest sensitive and tech savvy $\longrightarrow$ $\uparrow$ branch closures
\end{itemize}
\end{frame}

\begin{frame}{Openings}
    \textbf{Construct a bank-zip-year panel}\\
    \vspace{0.5cm}
    \textbf{Dependent variable: indicator if a bank enters into a zip code in a given year}\\
    \vspace{0.5cm}
    \textbf{Candidate zip-codes for opening}\\
    \begin{itemize}
        \item Zips in CBSAs where bank owned a branch at t-1
        \item Zips in CBSAs where bank opened a branch at t
        \item Drop all zips with no branches ever
